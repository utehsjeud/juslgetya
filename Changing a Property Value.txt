<h2>JavaScript Object.defineProperty()</h2>

<p>This example changes a property value.</p>

<p id="demo"></p>

<script>
// Create an object:
var person = {
  firstName: "Sanu",
  lastName : "santh",
  language : "EN" 
};
// Change a property
Object.defineProperty(person, "language", {value:"NO"})
document.getElementById("demo").innerHTML = person.language;
</script>
